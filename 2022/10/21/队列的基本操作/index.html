队列的基本操作

#include<stdio.h>
#include<malloc.h>
#define OK 1
#define ERROR 0
#define TRUE 1
#define FALSE 0
#define OVERFLOW -2
typedef struct QNode{
int data;
struct QNode *next;
}QNode,*QueuePtr;
typedef struct{
QueuePtr front;
QueuePtr rear;
}LinkQueue;
int visit(int c)
{
printf(“%d “,c);
return 1;
}
int InitQueue(LinkQueue &Q);
int EnQueue(LinkQueue &Q,int e);
int DeQueue(LinkQueue &Q,int &e);
int QueueEmpty(LinkQueue Q);
int QueueLength(LinkQueue Q);
int QueueTraverse(LinkQueue Q,void visit());

int InitQueue(LinkQueue &Q)
{
Q.front=Q.rear=(QNode*)malloc(sizeof(QNode));
if(!Q.front) return ERROR;
Q.front->next=NULL;
return OK;
}

int EnQueue(LinkQueue &Q,int e)
{

QueuePtr p=(QueuePtr)malloc(sizeof(QNode));
if(!p) return ERROR;
p->data=e;p->next=NULL;
Q.rear->next=p;
Q.rear=p;
return OK;

}

int DeQueue(LinkQueue &Q,int &e)
{
if(Q.front==Q.rear) return ERROR;
QueuePtr p;
p=Q.front->next;
e=p->data;
Q.front->next=p->next;
if(Q.rear==p) Q.rear=Q.front;
free(p);
return OK;
}

int QueueEmpty(LinkQueue Q)
{
if(Q.front==Q.rear)
return -1;
else
return 1;
}

int QueueLength(LinkQueue Q)
{
if(Q.front==Q.rear) return ERROR;
int i=0;
while(Q.front!=Q.rear)
{
i++;
Q.front=Q.front->next;
}
return i;
}

int QueueTraverse(LinkQueue Q,int (*visit)(int))
{
QueuePtr P;
P=Q.front->next;
while(P)
{
visit(P->data);
P=P->next;
}
return OK;
}

int main()
{
LinkQueue Q;
InitQueue(Q);
int i;
for(i=1;i<=6;i++)
{
EnQueue(Q,2*i-1);
}
QueueTraverse(Q,visit);
printf(“\n”);
EnQueue(Q,7);
QueueTraverse(Q,visit);
printf(“\n”);
int e;
DeQueue(Q,e);
QueueTraverse(Q,visit);
printf(“\n”);
int k,m;
k=QueueLength(Q);
printf(“%d\n”,k);
m=QueueEmpty(Q);
if(m>0)
printf(“isnotEmpty”);
else
printf(“isEmpty”);
return 0;
}

C语言链表及测试

#include<stdio.h>
#include<malloc.h>
#define OK 1
#define ERROR 0
#define TRUE 1
#define FALSE 0
#define OVERFLOW -2

typedef struct LNode
{
int date;
struct LNode *next;
}LNode,*LinkList;

int InitList(LinkList *L);
int DestroyList(LinkList *L);
int ClearList(LinkList *L);
int ListEmpty(LinkList L);
int ListLength(LinkList L);
int GetElem(LinkList L,int i,int *e);
int LocateElem(LinkList L,int e);
int PriorElem(LinkList L,int cur_e,int *pre_e);
int NextElem(LinkList L,int cur_e,int *next_e);
int ListInsert(LinkList *L,int i,int e);
int ListDelete(LinkList *L,int i,int *e);
int ListTraverse(LinkList L);

int InitList(LinkList *L)
{
    *L=(LinkList)malloc(sizeof(LNode));
    LinkList *t=L;
    for(int i=1;i<5;i++)
    {
        LinkList *a=(LinkList*)malloc(sizeof(LNode));
        (*a)->date=i;
        (*a)->next=NULL;
        (*t)->next=(*a);
        (*t)=(*t)->next;
    }
}		

int DestroyList(LinkList *L)
{
    LinkList p;
    while(*L)
    {
        p=*L;
        *L=(*L)->next;
        free(p);
    }
    return OK;
}

int ClearList(LinkList *L)
{
    LinkList p,q;
    p=(*L)->next;
    (*L)->next=NULL;
    while(p)
    {
        q=p;p=p->next;
        free(q);
    }
    return OK;
}

int ListEmpty(LinkList L)
{
    if(L->next)
        return FALSE;
    else
        return TRUE;
}

int ListLength(LinkList L)
{
    LinkList p;
    int i=0;
    p=L->next;
    while(p)
    {
        i++;
        p=p->next;
    }
    return i;
}

int GetElem(LinkList L,int i,int *e)
{
    int j=i;
    LinkList p;
    p=L->next;
    while(p && j<i)
    {
        j++;
        p=p->next;
    }
    if(!p || j>i)
        return ERROR;
    *e=p->date;
    return OK;
}

int LocateElem(LinkList L,int e)
{
    int i=0;
    LinkList p;
    p=L->next;
    while(p)
    {
        i++;
        if(p->date==e)
        return i;
        p=p->next;
    }
    return 0;
}

int PriorElem(LinkList L,int cur_e,int *pre_e)
{
    LinkList p,q;
    if(L->next && L->next->date==cur_e)
        return ERROR;
    q=L->next;
    p=q->next;
    while(p && p->date!=cur_e)
    {
        q=p;
        p=p->next;
    }
    if(!p)
        return ERROR;
    *pre_e=q->date;
    return OK;
}

int NextElem(LinkList L,int cur_e,int *next_e)
{
    LinkList p;
    p=L->next;
    while(p && p->date!=cur_e)
        p=p->next;
    if(!p)
        return ERROR;
        *next_e=p->next->date;
        return OK;
}

int ListInsert(LinkList *L,int i,int e)
{
    int j=0;
    LinkList p,s;
    p=*L;
    while(p && j<i-1)
    {
        j++;
        p=p->next;
    }
    if(!p || j>i-1)
        return ERROR;
    s=(LinkList)malloc(sizeof(LNode));
    s->date=e;
    s->next=p->next;
    p->next=s;
    return OK;
}

int ListDelete(LinkList *L,int i,int *e)
{
    LinkList p,q;
    int j;
    p=*L;
    j=0;
    while(p->next && j<i-1)
    {
        p=p->next;
        ++j;
    }
    if(!(p->date) || j>i-1)
        return ERROR;
    q=p->next;
    p->next=q->next;
    *e=q->date;
    free(q);
    return OK;
}

int ListTraverse(LinkList L)
{
    LinkList p;
    p=L->next;
    while(p)
    {
        printf("%d",p->date);
        p=p->next;
    }
    printf("\n");
    return OK;
}

{
LinkList L;
L=(LinkList)malloc(sizeof(LNode));
LinkList P;
P=L;
int i;
for(i=1;i<=6;i++)
{
P->next=(LinkList)malloc(sizeof(LNode));
P=P->next;
P->date=i*2-1;
}
P->next=NULL;
ListTraverse(L);
int e;
ListDelete(&L,2,&e);
ListTraverse(L);
ListInsert(&L,3,6);
ListTraverse(L);
int next_e;
NextElem(L,5,&next_e);
printf(“%d\n”,next_e);
int pre_e;
PriorElem(L,11,&pre_e);
printf(“%d\n”,pre_e);
int a;
a=LocateElem(L,7);
printf(“%d\n”,a);
int len;
len=ListLength(L);
printf(“%d\n”,len);
if(!ListEmpty(L))
if(ClearList(&L))
{
printf(“OK”);
}
else
printf(“Error!”);
else
printf(“Error!”);
return 0;
}
